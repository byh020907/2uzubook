CREATE TABLE `major` (
    `id` INT(1) NOT NULL PRIMARY KEY AUTO_INCREMENT,
    `name` VARCHAR(10) NOT NULL UNIQUE
);

CREATE TABLE `keyword` (
    `id` INT(8) NOT NULL PRIMARY KEY AUTO_INCREMENT,
    `name` VARCHAR(64) NOT NULL UNIQUE
);

CREATE TABLE `user` (
    `adm_year` INT(4) NOT NULL,
    `stu_id` INT(5) NOT NULL,
    `id` VARCHAR(32) NOT NULL PRIMARY KEY,
    `password` CHAR(64) NOT NULL,
    `intro` TEXT DEFAULT NULL,
    `name` VARCHAR(32) NOT NULL,
    `gender` CHAR(1) NOT NULL,
    `major` INT(1) NOT NULL,
    `email` VARCHAR(64) NOT NULL UNIQUE,
    FOREIGN KEY (`major`) REFERENCES `major`(`id`)
);

CREATE TABLE `award` (
    `id` INT(8) NOT NULL PRIMARY KEY AUTO_INCREMENT,
    `user` VARCHAR(32) NOT NULL,
    `name` VARCHAR(64) NOT NULL,
    `ins` VARCHAR(32) NOT NULL,
    `date` DATE NOT NULL,
    `keyword` VARCHAR(32) NOT NULL,
    FOREIGN KEY (`user`) REFERENCES `user`(`id`)
);

CREATE TABLE `cert` (
    `id` INT(8) NOT NULL PRIMARY KEY AUTO_INCREMENT,
    `user` VARCHAR(32) NOT NULL,
    `name` VARCHAR(64) NOT NULL,
    `ins` VARCHAR(32) NOT NULL,
    `date` DATE NOT NULL,
    `keyword` VARCHAR(32) NOT NULL,
    FOREIGN KEY (`user`) REFERENCES `user`(`id`)
);

CREATE TABLE `project` (
    `id` INT(8) NOT NULL PRIMARY KEY AUTO_INCREMENT,
    `user` VARCHAR(32) NOT NULL,
    `name` VARCHAR(64) NOT NULL,
    `desc` TEXT NOT NULL,
    `startdate` DATE NOT NULL,
    `enddate` DATE,
    `keyword` VARCHAR(32) NOT NULL,
    FOREIGN KEY (`user`) REFERENCES `user`(`id`)
);

CREATE TABLE `club` (
    `id` INT(8) NOT NULL PRIMARY KEY AUTO_INCREMENT,
    `user` VARCHAR(32) NOT NULL,
    `name` VARCHAR(64) NOT NULL,
    `desc` TEXT NOT NULL,
    `startdate` DATE NOT NULL,
    `enddate` DATE,
    `keyword` VARCHAR(32) NOT NULL,
    FOREIGN KEY (`user`) REFERENCES `user`(`id`)
);

CREATE TABLE `company` (
    `id` INT(8) NOT NULL PRIMARY KEY AUTO_INCREMENT,
    `name` VARCHAR(64) NOT NULL,
    `email` varchar(64)NOT NULL,
    `key` CHAR(5),
    `keyword` VARCHAR(32) NOT NULL,
     unique key email (email)
);

CREATE TABLE `etc_category` (
    `id` INT(1) NOT NULL PRIMARY KEY AUTO_INCREMENT,
    `name` VARCHAR(10) NOT NULL UNIQUE
);

CREATE TABLE etc(
	id int(1) NOT NULL PRIMARY KEY KEY AUTO_INCREMENT,
    user varchar(32) not null,
	title varchar(64) not null,
    content text not null,
    date DATE not null,
    category_num int(1) not null,
    FOREIGN KEY (`category_num`) REFERENCES `etc_category`(`id`),
    FOREIGN KEY (`user`) REFERENCES `user`(`id`)
);
